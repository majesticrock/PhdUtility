485  {

488  // This comparison is well defined because we save the momentum as char i.e. byte
489  // which is easily comparable
490  if (delta.first.momentum_list[0].name < delta.second.momentum_list[0].name) {
491  std::swap(delta.first, delta.second);
492  if (delta.first.momentum_list[0].factor < 0) {
493  delta.first.flip_momentum();
494  delta.second.flip_momentum();
495  }
496  if (delta.first.add_Q) {
497  delta.first.add_Q = false;
498  delta.second.add_Q = !(delta.second.add_Q);
499  }
500  }
501  for (auto& op : operators) {
502  op.momentum.replace_occurances(delta.first.momentum_list[0].name, delta.second);
503  }

505  coeff.momenta.replace_occurances(delta.first.momentum_list[0].name, delta.second);
506  }
507  }
508  }
509 
510  for (auto& op : operators) {
511  if (op.type == CDW_Type && op.momentum.add_Q) {
512  op.momentum.add_Q = false;
513  op.is_daggered = !(op.is_daggered);
514  }
515  }
516 
517  for (int i = 0U; i < operators.size(); ++i)
518  {
519  for (int j = i + 1U; j < operators.size(); ++j)
520  {
521  if (operators[i].type > operators[j].type) {
522  std::swap(operators[i], operators[j]);
523  }
524  else if (operators[i].type == operators[j].type) {
525  if (momentum_order(operators[i].momentum, operators[j].momentum)) {
526  std::swap(operators[i], operators[j]);
527  }

537  if (momentum.momentum_list[0].factor < 0) {

572  if (coeff.inversion_symmetry) momentum.flip_momentum();
