606  void normal_order(std::vector<Term>& terms) {
607  for (int t = 0; t < terms.size();) {
608  normal_order_outerLoop:
609  if (t >= terms.size()) break;
610  size_t n = terms[t].operators.size();
611  size_t new_n{};
612  // First sort so that the bosons are upfront
613  while (n > 1U) {
614  new_n = 0U;
615  for (size_t i = 1U; i < terms[t].operators.size(); ++i)
616  {
617  if (terms[t].operators[i - 1].is_fermion && !terms[t].operators[i].is_fermion) {
618  new_n = i;
619  std::swap(terms[t].operators[i - 1], terms[t].operators[i]);
620  }
621  }
622  n = new_n;
623  }
624 
625  n = terms[t].operators.size();
626  new_n = 0U;
627  while (n > 1U) {
628  new_n = 0U;
629  for (size_t i = 1U; i < terms[t].operators.size(); ++i)
630  {
631  if (!terms[t].operators[i - 1].is_fermion && terms[t].operators[i].is_fermion) continue;
632  if (!(terms[t].operators[i - 1].is_daggered) && (terms[t].operators[i].is_daggered)) {
633  bool other_deltas = false;
634  new_n = i;
635  // Swap cc^+
636  std::swap(terms[t].operators[i - 1], terms[t].operators[i]);
637 
638  // Add a new term where cc^+ is replaced by the appropriate delta
639  Term new_term(terms[t]);
640  // flip the signs if we have fermions
641  if (terms[t].operators[i - 1].is_fermion && terms[t].operators[i].is_fermion) {
642  terms[t].flip_sign();
643  }
644  if (new_term.operators[i - 1].indizes.size() != new_term.operators[i].indizes.size()) {
645  throw std::invalid_argument("Operators do not have the same index count.");
646  }
647 
648  if( (new_term.operators[i - 1].first_index() == Index::SpinUp && new_term.operators[i].first_index() == Index::SpinDown)
649  || (new_term.operators[i - 1].first_index() == Index::SpinDown && new_term.operators[i].first_index() == Index::SpinUp) ) {
650  continue;
651  }
652  else if(new_term.operators[i - 1].first_index() != new_term.operators[i].first_index()) {
653  new_term.delta_indizes.push_back(
654  make_delta(new_term.operators[i - 1].first_index(), new_term.operators[i].first_index()));
655  }
656  for (int c = 1; c < new_term.operators[i - 1].indizes.size(); c++)
657  {
658  // if the indizes are not the same we emplace a delta
659  // otherwise no action is required
660  if (new_term.operators[i - 1].indizes[c] != new_term.operators[i].indizes[c]) {
661  other_deltas = true;
662  new_term.delta_indizes.push_back(
663  make_delta(new_term.operators[i - 1].indizes[c], new_term.operators[i].indizes[c]));
664  }
665  }
666  if (new_term.operators[i - 1].momentum != new_term.operators[i].momentum) {
667  other_deltas = true;
668  new_term.delta_momenta.push_back(
669  make_delta(new_term.operators[i - 1].momentum, new_term.operators[i].momentum)
670  );
671  }
672  else {
673  other_deltas = true;
674  }
675 
676  new_term.operators.erase(new_term.operators.begin() + i - 1, new_term.operators.begin() + i + 1);
677  if (other_deltas) terms.push_back(new_term);
678  }
679  else if (terms[t].operators[i - 1] == terms[t].operators[i]) {
680  if (terms[t].operators[i - 1].is_fermion) {
681  // two identical fermion operators = 0
682  terms.erase(terms.begin() + t);
683  goto normal_order_outerLoop;
684  }
685  }
686  }
687  n = new_n;
688  }
689  ++t;
690  }
691  }
